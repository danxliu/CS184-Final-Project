  \begin{minipage}{0.78\paperwidth}
  \raggedright
  \decklabel{02 -- Core energy}\par
  \vspace{0.54cm}
  \slidetitle{An energy that becomes infinite at self-contact.}
  \thinrule
  \slidebody{Our discrete tangent-point energy sums a kernel over triangle pairs. If two distant patches approach zero separation, the kernel diverges.}
  \vspace{0.35cm}
  {\fontsize{10.5}{12.5}\selectfont\color{Dim}
  \[
    \widehat{\Phi}(x)
      = \sum_{t_1 \ne t_2}
        a_1 a_2
        \frac{|\langle n_1, c_1 - c_2\rangle|^6}
             {\|c_1 - c_2\|^{12}}
  \]
  }
  \end{minipage}
\end{frame}

  {\fontsize{10.3}{13}\selectfont\color{Dim}\raggedright

  \par}
